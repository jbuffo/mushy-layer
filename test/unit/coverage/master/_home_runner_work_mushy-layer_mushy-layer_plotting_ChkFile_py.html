278 def compute_diagnostic_vars(self): 

279 # First get parameters 

280 

281 if self.inputs is None: 

282 print('No inputs file available, so do not know parameters and cannot compute diagnostic variables') 

283 sys.exit(-1) 

284 

285 conc_ratio = float(self.inputs['parameters.compositionRatio']) 

286 stefan = float(self.inputs['parameters.stefan']) 

287 cp = float(self.inputs['parameters.specificHeatRatio']) 

288 pc = float(self.inputs['parameters.waterDistributionCoeff']) 

289 theta_eutectic = 0.0 

290 

291 enthalpy = self.get_data('Enthalpy') 

292 bulk_salinity = self.get_data('Bulk concentration') 

293 

294 enthalpy_eutectic = np.empty(enthalpy.shape) 

295 enthalpy_solidus = np.empty(enthalpy.shape) 

296 enthalpy_liquidus = np.empty(enthalpy.shape) 

297 

298 porosity = np.empty(enthalpy.shape) 

299 temperature = np.empty(enthalpy.shape) 

300 liquid_salinity = np.empty(enthalpy.shape) 

301 solid_salinity = np.empty(enthalpy.shape) 

302 

303 # Now compute bounding energies 

304 cols = enthalpy.shape[0] 

305 rows = enthalpy.shape[1] 

306 for j in range(cols): 

307 

308 for i in range(rows): 

309 eutectic_porosity = (conc_ratio + bulk_salinity[j, i]) / (theta_eutectic + conc_ratio) 

310 enthalpy_eutectic[j, i] = eutectic_porosity * (stefan + theta_eutectic * (1 - cp)) + cp * theta_eutectic 

311 enthalpy_solidus[j, i] = cp * (theta_eutectic + max(0.0, (-bulk_salinity[j, i] - conc_ratio) / pc)) 

312 enthalpy_liquidus[j, i] = stefan - bulk_salinity[j, i] + theta_eutectic + theta_eutectic 

313 

314 # Compute diagnostic variables 

315 for j in range(cols): 

316 

317 for i in range(rows): 

318 if enthalpy[j, i] <= enthalpy_solidus[j, i]: 

319 porosity[j, i] = 0.0 

320 temperature[j, i] = enthalpy[j, i] / cp 

321 liquid_salinity[j, i] = 0.0 

322 solid_salinity[j, i] = bulk_salinity[j, i] 

323 elif enthalpy[j, i] <= enthalpy_eutectic[j, i]: 

324 porosity[j, i] = (enthalpy[j, i] - theta_eutectic * cp) / (stefan + theta_eutectic * (1 - cp)) 

325 temperature[j, i] = theta_eutectic 

326 liquid_salinity[j, i] = theta_eutectic 

327 solid_salinity[j, i] = bulk_salinity[j, i] / (1 - porosity[j, i]) 

328 elif enthalpy[j, i] < enthalpy_liquidus[j, i]: 

329 a = conc_ratio * (cp - 1) + stefan * (pc - 1) 

330 b = conc_ratio * (1 - 2 * cp) + enthalpy[j, i] * (1 - pc) \ 

331 - bulk_salinity[j, i] * (cp - 1) - pc * stefan 

332 c = (bulk_salinity[j, i] + conc_ratio) * cp + pc * enthalpy[j, i] 

333 porosity[j, i] = (-b - np.sqrt(b * b - 4 * a * c)) / (2 * a) 

334 

335 liquid_salinity[j, i] = (bulk_salinity[j, i] + conc_ratio * (1 - porosity[j, i])) / ( 

336 porosity[j, i] + pc * (1 - porosity[j, i])) 

337 temperature[j, i] = -liquid_salinity[j, i] 

338 solid_salinity[j, i] = (pc * bulk_salinity[j, i] - conc_ratio * porosity[j, i]) / ( 

339 porosity[j, i] + pc * (1 - porosity[j, i])) 

340 else: 

341 porosity[j, i] = 1.0 

342 temperature[j, i] = enthalpy[j, i] - stefan 

343 liquid_salinity[j, i] = bulk_salinity[j, i] 

344 solid_salinity[j, i] = 0.0 

345 

346 # Finally, save data into boxes 

347 self.save_data('Porosity', porosity) 

348 self.save_data('Temperature', temperature) 

349 self.save_data('LiquidSalinity', liquid_salinity) 

350 self.save_data('SolidSalinity', solid_salinity)
